41 const int maxPowerTen = 38;
42 // == std::numeric_limits<float>::max_exponent10;
43 // == FLT_MAX_10_EXP in float.h, so can represent FLT_MAX = 3.402823466e+38f
44 // to FLT_MIN_10_EXP in float.h, so can represent FLT_MIN = 1.175494351e-38f
45 // This is chosen to limit the size of the power-of-ten lookup table below,
46 // and make code the same if store value in a 32-bit double.
47 // For real-life data, value of a physical measurement unit is unlikely to exceed 10^38,
48 // despite DBL_MAX_10_EXP == 308.
