00003 enum unc_types
00004 {   // 16 type bits used by UReal uncTypes. Bit set = 1 means a positive attribute.
00005             VALUE_ZERO = 1 << 0, // Too near zero to allow relsd = sd /mean
00006             // A few times std::numeric_limits<float>::min() is a suitable limit?
00007             VALUE_INTEGER = 1 << 1, // Exact integer 2,3,4 ... may not be possible to store.
00008             VALUE_RATIONAL = 1 << 2, // Integers & Fractions 1/2, 1/3, 2/3, 5/4 ...
00009             // so don't need irrational - must be if not rational.
00010             // Irrational like pi, e ... representated as accurately as possible.
00011             VALUE_NEGATIVE_ONLY = 1 << 3, // Value can ONLY be < 0.. ? or <=0?
00012             VALUE_POSITIVE_ONLY = 1 << 4, // Value can ONLY be >=0..
00013             UNC_KNOWN = 0x20,    // Uncertainty known (but if not value may known OK).
00014             UNC_NOPLUS = 0x40, // Uncertainty can only be < value, + = zero.
00015             UNC_NOMINUS = 0x80, // Uncertainty can only be > value - = zero.
00016             // Integer, Rational or Exact by definition like 25.4 mm per inch,
00017             // if unc = 0.0f or both UNC_NOPLUS & UNC_NOMINUS.
00018             UNC_QUAN_DECIMAL = 0X100, // Quantised by least significant decimal digit.
00019             UNC_QUAN_BINARY = 0X200, // Quantised by least significant binary digit.
00020             UNC_EXPLICIT = 0x400, //If bit set = 1, uncertainty implicit from sig digits.
00021             // or explicit if +/- or calculated (normal default case, bit = 0).
00022             UNC_RECTANGULAR = 0x800, // Uncertainty has rectangular probability distribution.
00023             UNC_TRIANGULAR = 0x1000, // Uncertainty has triangular probability distribution.
00024             //  If not rectangular or triangular, then normal or Gaussian.
00025             // But is triangular & gaussian different in real life?
00026             DEG_FREE_EXACT = 0x2000, // Known number of observations -1,
00027             // so default zero means 1 observation.
00028             // else estimated, for example, using Welch-Satterthwaite.
00029             DEG_FREE_KNOWN = 0x4000, // Degrees of freedom defined.
00030             SPARE = 0x8000 // This unc_type is spare.
00031 };

00048             addSISymbol = 0x10,  // bit 4 = 1, add suffix SI symbol G, M, k, m, u, n, p ...
00049             // If one is applicable, else = 0 do nothing.
00050             addSIPrefix = 0x20,  // bit 5 = 1, add suffix SI prefix Giga, Mega, kilo ...
00051             noisyDigit = 0x40, // Add extra 'noisy' decimal digit to value & sd
00052             // This means is suitable for input to other calculations because the
00053             // random extra 2 to 3 bits will approximate a continuous function.
00054             // Quantisation to fewer digits would distort statistical calculations.
00055             // Default is to use minimum decimal digits for clearer display.
