112 {
113 public:

118  template<class CharT, class Traits, class Range>
119  void print(const Range& r, std::basic_ostream<CharT, Traits>& os) const {
120  // void print(std::basic_ostream<CharT, Traits>& os, const Range& r) const {
121  // Capture the print arguments.
122  // Range iterators.
123  typename boost::range_iterator<const Range>::type
124  first(boost::begin(r)),
125  last(boost::end(r));
126  // Assemble requirements into tuple: ostream, and range pair of iterators.
127  tuple<requirements, _os&, _iter, _iter> args(os, first, last);
128  // and forward to the real implementation.
129  do_print(get<0>(args), get<1>(args), get<2>(args));
130  // get<0>() is ostream, get<1>() is first, get<2>() is last.
131  }
132  virtual ~abstract_printer() {}
133 protected:
134  // Define the requirements in an mpl::vector for containers to be printed.
135  typedef boost::mpl::vector<
136  base_and_derived<std::ios_base, _os>, // _os must derive from std::ios_base.
137  ostreamable<_os, _t>, // type _t must be ostreamable.
138  ostreamable<_os, const char*>, // C string must be ostreamable.
139  forward_iterator<_iter, const _t&>, // t range must have forward iterator.
140  same_type<_t, forward_iterator<_iter, const _t&>::value_type> // _t's type value and iterator must match.
141  > requirements;
142 
143  // Use type_erasure::any to enforce the requirements onto ostream and iterator.
144  typedef boost::type_erasure::any<requirements, _os&> ostream_type;
145  typedef boost::type_erasure::any<requirements, _iter> iterator_type;
146  // Declare the pure virtual function `do_print` that is defined in all real printers.
147  virtual void do_print(
148  ostream_type os, iterator_type first, iterator_type last
149  ) const = 0;
150 }; // class abstract_printer

174 class decor_printer : public abstract_printer
175 {
176 public:
177  explicit decor_printer(
178  std::size_t num_columns = 0,
179  std::size_t wid = 0,
180  const std::string& pre = "\n",
181  const std::string& sep = " ",
182  const std::string& suf = "\n",
183  const std::string& term = "\n")
184  :
185  cols(num_columns), width(wid), prefix(pre), separator(sep), terminator(term), suffix(suf)
186  {
194  }
195 protected:
196  // Print all items in rows.
197  virtual void do_print(
198  ostream_type os, iterator_type first, iterator_type last) const
199  {
200  os << prefix.c_str();
201  std::size_t count = 0;
202  for(; first != last; ++first)
203  {
204  if (width > 0)
205  {
206  static_cast<std::ios_base&>(os).width(width);
207  }
208  os << *first;
209  boost::type_erasure::any<requirements, _iter> temp = first;
210  ++temp;
211  if(temp != last)
212  {
213  os << separator.c_str();
214  }
215  ++count;
216  if((cols != 0) && (count % cols == 0))
217  { // Last column.
218  os << suffix.c_str(); // Usually newline at end of a row.
219  }
220  }
221  os << terminator.c_str(); // Perhaps useful to have a terminating newline?
222  }
223 private:
224  std::size_t cols; // Set by constructor, number of columns (default 10).
225  std::size_t width; // Set by constructor, optional column width (default zero).
226  std::string prefix; // Set by constructor, for example, "{ ".
227  std::string separator; // Set by constructor, for example, ", ".
228  std::string suffix; // Set by constructor, for example "\n".
229  std::string terminator; // Set by constructor, for example, "}\n".
230 }; // class decor_column_printer
