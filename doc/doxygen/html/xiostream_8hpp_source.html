131 // Applicator is a class that stores a pointer to a function that takes
132 // a reference to an std::ios_base (or derived translator) argument, and
133 // an argument of the type for which the applicator is parameterized.
134 // Applicator classes have the function call operator overloaded
135 // so as to simulate a function call with the argument of the parameter type.
136 // Use, for example: oapp<int> stars(_stars); // allows << stars(5) ...
137 // where function is std::ostream& _stars(std::ostream&, int);

139 // Global ostream applicator using template oapp instantiated for type int,
140 // & initialised with the address of function with one int parameter.
141 // Possible to use oapp<int> stars(_stars); which allows << stars(5) ...
142 // but instead stars & chars done a simpler way, see S Teale p 181-3.
143 template<typename T> class oapp // Applicator for ostream.
144 {
145 public:
146  oapp(std::ostream& (*f)(std::ostream&, T)) : func(f)
147  {
148  };

150  {
151  return omanip<T>(func, v);
152  }
153 private:

210 void outIOstates(std::ios_base::iostate rdState, std::ostream& os, const char* terminator)
211 {
212  std::ios_base::fmtflags const savedflags = os.flags(); // Save to restore.
213  // Clear any unused and invalid bits in rdState.
214  rdState &= std::ios_base::goodbit | std::ios_base::eofbit | std::ios_base::failbit | std::ios_base::goodbit;
215  // enum _Iostate {goodbit = 0x0, eofbit = 0x1, failbit = 0x2, badbit = 0x4, _Statmask = 0x7};
216  // but use std::ios_base::goodbit; for portability.
217  // MSVC/Dinkumware defines _Statmask 0x17, _Hardfail also added recently.
218  os << "rdState ("<< std::showbase << std::hex << rdState << ") " << std::dec;
219  if (rdState == std::ios_base::goodbit)
220  {
221  os << "good"; // 1st so no leading space.
222  }
223  else
224  { // At least one bit set, so all need a space first.
225  if ((rdState & std::ios_base::eofbit) != 0)
226  {
227  os << " eof";
228  }
229  if ((rdState & std::ios_base::failbit) != 0)
230  {
231  os << " fail";
232  }
233  if ((rdState & std::ios_base::badbit) != 0)
234  {
235  os << " bad";
236  }
237  }
238  os << terminator; // eg "\n" or ", " or " ".
239  os.flags(savedflags); // Restore.
240 } // outIOstates

279 void outFmtFlags(std::ios_base::fmtflags fmtFlags, std::ostream& os, const char* terminator)
280 {
281  const int count = 16; // because using unsigned short int for flags.
282  std::ios_base::fmtflags flags = os.flags(); // Save to restore.
283  fmtFlags &= static_cast<std::ios_base::fmtflags>(0x7FFF); // _Fmtmask // clear un-used bits.
284  os << "iosFormatFlags (" << std::showbase << std::hex << fmtFlags << std::dec << ")"; // hex value.
285  if (fmtFlags != 0)
286  {
287  for (int i = 0, j = 1; i < count; ++i)
288  {
289  if ((fmtFlags & j) != 0)
290  {
291  os << ' ' << fmtFlagWords[i];
292  }
293  j <<= 1;
294  }
295  }
296  os << terminator; // eg "\n" or ". "
297  os.flags(flags); // Restore.
298 } // outFmtFlags

318 { // Return state and flags to same default state as when initialised.
