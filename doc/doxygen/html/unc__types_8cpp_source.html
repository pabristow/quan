24 enum unc_types
25 { // 16 type bits used by UReal uncTypes. Bit set = 1 means a positive attribute.
26  VALUE_ZERO = 1 << 0, // bit zero = 1 means value is zero.
27  // (Too near zero to allow relsd = sd /mean.)
28  // A few times std::numeric_limits<float>::min() may be a suitable limit?
29  VALUE_INTEGER = 1 << 1, // Exact integer 2,3,4 ... may not be possible to store.
30  VALUE_RATIONAL = 1 << 2, // Integers & Fractions 1/2, 1/3, 2/3, 5/4 ...
31  // so don't need irrational - must be if not rational.
32  // Irrational like pi, e ... representated as accurately as possible.
33  VALUE_NEGATIVE_ONLY = 1 << 3, // Value can ONLY be < 0.. ? or <=0?
34  VALUE_POSITIVE_ONLY = 1 << 4, // Value can ONLY be >=0..
35  UNC_KNOWN = 1 << 5, // Uncertainty known (but if not value may known OK).
36  UNC_NOPLUS = 1 << 6, // Uncertainty can only be < value, + = zero.
37  UNC_NOMINUS = 1 << 7, // Uncertainty can only be > value - = zero.
38  // Integer, Rational or Exact by definition like 25.4 mm per inch,
39  // if unc = 0.0f or both UNC_NOPLUS & UNC_NOMINUS.
40  UNC_QUAN_DECIMAL = 1 << 8, // Quantised by least significant decimal digit.
41  UNC_QUAN_BINARY = 1 << 9, // Quantised by least significant binary digit.
42  UNC_EXPLICIT = 1 << 10, // If bit set = 1, uncertainty implicit from read significant digits,
43  // for example: uncertainRead of 10.0 will set this bit and set uncertainty = 0.05,
44  // or explicit if +/- or calculated (normal default case, bit = 0).
45  UNC_RECTANGULAR = 1 << 11, // Uncertainty has rectangular probability distribution.
46  UNC_TRIANGULAR = 1 << 12, // Uncertainty has triangular probability distribution.
47  // If not rectangular or triangular, then normal or Gaussian.
48  // But is triangular & Gaussian different in real life?
49  DEG_FREE_EXACT = 1 << 13, // Known number of observations -1,
50  // so default zero means 1 observation.
51  // else estimated, for example, using Welch-Satterthwaite.
52  DEG_FREE_KNOWN = 1 << 14, // Degrees of freedom defined.
53  SPARE = 1 << 15 // This last unc_type is spare.
54 };

56 enum uncertainflags
57 { // For control of printing, similar to ios flags.
58  none = 0, // Default.
59  firm = 1 << 0, // bit 0: == 0 == false means flex, or firm == 1 == true.
60  setScaled = 1 << 1, // bit 1 Set scaled == 1 or not scaled == 0.
61  // by fixed factor SetScale and << scale ...
62  // power10 in range -max to +max.
63  autoScaled = 1 << 2, // bit 2: auto = 1 with << autoscale ...
64  plusMinus = 1 << 3, // bit 3 = 1 means add +/- uncertainty << plusminus ...
65  addSISymbol = 1 << 4, // bit 4 = 1, add suffix SI symbol G, M, k, m, u, n, p ...
66  // If one is applicable, else = 0 do nothing.
67  addSIPrefix = 1 << 5, // bit 5 = 1, add suffix SI prefix Giga, Mega, kilo ...
68  noisyDigit = 1 << 6, // Add extra 'noisy' decimal digit to value & sd
69  // This means is suitable for input to other calculations because the
70  // random extra 2 to 3 bits will approximate a continuous function.
71  // Quantization to fewer digits would distort statistical calculations.
72  // Default is to use minimum decimal digits for clearer display.
73  // Extra digit also added when degrees of freedom > 10.
74  useSetSigDigits = 1 << 7, // if bit 7 = 1 use set sig digits else calculate from sd(isUseSetSigDigits).
75  useSetUncSigDigits = 1 << 8 // if bit 8 = 1 use set Unc sigfig (isStdDevSigDigitsSet).
76  // else calculate from degrees of freedom.
77 }; // enum uncertainflags
