00973   unc<is_correlated>& operator+= (const unc<is_correlated>& ud)
00974   { // unc += unc
00975     if (is_correlated)
00976     {
00977       m_uncertainty += ud.m_uncertainty;  // If correlated then just add.
00978       m_degFree = // Choose degrees of freedom of the max uncertainty.
00979         (m_uncertainty > ud.m_uncertainty) ? m_degFree : ud.m_degFree;
00980       // Exact method of estimating not known, but is a reasonable approximation
00981       // provided really is highly correlated because less well predicted
00982       // variance can be predicted well with the better predicted.
00983       // See email from Steve LR Ellison, Lab of Government Chem, 24 Apr 98.
00984     }
00985     else
00986     { // Uncorrelated so Propagation of Uncertainty rules apply.
00987       // Barry N Taylor & Chris E Kuyatt, NIST Technical note 1297 (1994),
00988       // Guidelines for Evaluating & Expressing the Uncertainty of Measurements.
00989       // http:// physics.nist.gov/pubs/guidelines.appa.html
00990       // 'Combined uncertainty' Appendix A formula A-3 = "sqrt(sum of squares)"
00991       double comb_uncertainty = sqrtSumSqrs(m_uncertainty, ud.m_uncertainty);
00992       m_degFree =  // Welch-Satterthwaite - see NIST Appendix A3 & Appendix B-1.
00993         // Effective degrees of freedom <= - see Eq B-2, so use floor function.
00994         unsigned short ( // Degrees of freedom must be integer by definition.
00995         floor(
00996         pow4<double>(comb_uncertainty)/
00997         (pow4<double>(m_uncertainty)/m_degFree + pow4<double>(ud.m_uncertainty) / ud.m_degFree))
00998         );
00999       m_uncertainty = float(comb_uncertainty);
01000     }
01001     m_types &= ~DEG_FREE_EXACT; // This warns of an approximation - no longer exact.
01002     m_types &= ud.m_types; // Imposes rules for all additions of uncs.
01003     // zero + zero = zero, restrictions on pos & neg only only retained if both,
01004     // int + int is int, rational + rational is rational, int + rational is rational.
01005     // either or both plusminus same if BOTH.
01006     // Quantisation retains only if both.
01007     m_value += ud.m_value;
01008     return *this;
01009   }
