71 class abstract_printer
72 {
73 public:
74  // print - write a sequence to a std::ostream in a manner specific to the derived class.
75  //
76  // Requires: Range must be a Forward Range whose elements can be printed to os.
77  template<class CharT, class Traits, class Range>
78  void print(std::basic_ostream<CharT, Traits>& os, const Range& r) const {
79  // Capture the arguments
80  typename boost::range_iterator<const Range>::type
81  first(boost::begin(r)),
82  last(boost::end(r));
83  tuple<requirements, _os&, _iter, _iter> args(os, first, last);
84  // and forward to the real implementation
85  do_print(get<0>(args), get<1>(args), get<2>(args));
86  }
87  virtual ~abstract_printer() {}
88 protected:
89  // define the concept requirements of the arguments of
90  // print and typedef the any types.
91  typedef boost::mpl::vector<
92  base_and_derived<std::ios_base, _os>,
93  ostreamable<_os, _t>,
94  ostreamable<_os, const char*>,
95  forward_iterator<_iter, const _t&>,
96  same_type<_t, forward_iterator<_iter, const _t&>::value_type>
97  > requirements;
98  typedef boost::type_erasure::any<requirements, _os&> ostream_type;
99  typedef boost::type_erasure::any<requirements, _iter> iterator_type;
100 
101  // do_print - This method must be implemented by derived classes.
102  virtual void do_print(
103  ostream_type os, iterator_type first, iterator_type last) const = 0;
104 }; // class abstract_printer

106 // separator_printer - writes the elements of a sequence
107 // separated by a fixed string. For example, if
108 // the separator is ", " separator_printer produces
109 // a comma separated list.
110 class separator_printer : public abstract_printer {
111 public:
112  explicit separator_printer(const std::string& sep) : separator(sep) {}
113 protected:
114  virtual void do_print(
115  ostream_type os, iterator_type first, iterator_type last) const {
116  if(first != last) {
117  os << *first;
118  ++first;
119  for(; first != last; ++first) {
120  os << separator.c_str() << *first;
121  }
122  }
123  }
124 private:
125  std::string separator;
126 }; // class separator_printer

157 // aligned_column_printer - formats a sequence in columns
158 // reading down. For example, given the sequence
159 // { 1, 2, 3, 4, 5 }, aligned_column_printer might print
160 // 1 4
161 // 2 5
162 // 3
163 class aligned_column_printer : public abstract_printer {
164 public:
165  aligned_column_printer(std::size_t column_width, std::size_t num_columns)
166  : width(column_width),
167  cols(num_columns)
168  {}
169 protected:
170  virtual void do_print(
171  ostream_type os, iterator_type first, iterator_type last) const
172  {
173  if(first == last) return;
174  std::vector<iterator_type> column_iterators;
175 
176  // find the tops of the columns
177  std::size_t count = 0;
178  for(iterator_type iter = first; iter != last; ++iter) {
179  ++count;
180  }
181  std::size_t rows = (count + cols - 1) / cols;
182  count = 0;
183  for(iterator_type iter = first; iter != last; ++iter) {
184  if(count % rows == 0) {
185  column_iterators.push_back(iter);
186  }
187  ++count;
188  }
189 
190  iterator_type last_col = column_iterators.back();
191 
192  // print the full rows
193  while(column_iterators.back() != last) {
194  for(std::vector<iterator_type>::iterator
195  iter = column_iterators.begin(),
196  end = column_iterators.end(); iter != end; ++iter)
197  {
198  static_cast<std::ios_base&>(os).width(width);
199  os << **iter;
200  ++*iter;
201  }
202  os << "\n";
203  }
204 
205  // print the rows that are missing the last column
206  column_iterators.pop_back();
207  if(!column_iterators.empty()) {
208  while(column_iterators.back() != last_col) {
209  for(std::vector<iterator_type>::iterator
210  iter = column_iterators.begin(),
211  end = column_iterators.end(); iter != end; ++iter)
212  {
213  static_cast<std::ios_base&>(os).width(width);
214  os << **iter;
215  ++*iter;
216  }
217  os << "\n";
218  }
219  }
220  }
221 private:
222  std::size_t width;
223  std::size_t cols;
224 }; // aligned_column_printer

299  const std::string& pre = "\n",

303  :

305  {
313  }

315  // Print all items in rows.
316  // was virtual void do_print(iterator_type first, iterator_type last, ostream_type os) const
317  virtual void do_print(ostream_type os, iterator_type first, iterator_type last) const
318  {
319  os << prefix.c_str();
320  std::size_t count = 0;
321  for(; first != last; ++first)
322  {
323  if (width > 0)

326  }
