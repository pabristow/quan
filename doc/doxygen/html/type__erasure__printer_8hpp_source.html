103 template< class CharT = char, class Traits = std::char_traits<char> >
104 class abstract_printer
105 {
106 public:
110 
111  template<class Range, class CharT, class Traits>
112  void print(const Range& r, std::basic_ostream<CharT, Traits>& os = std::cout) const
113  {
114  // Capture the print arguments.
115  // Range iterators.
116  typename boost::range_iterator<const Range>::type
117  first(boost::begin(r)),
118  last(boost::end(r));
119  // Assemble requirements into tuple: range pair of iterators and ostream.
120  tuple<requirements, _iter, _iter, _os&> args(first, last, os);
121  // and forward to the real implementation.
122  do_print(get<0>(args), get<1>(args), get<2>(args));
123  // get<0>() is first, get<1>() is last, get<2>() is ostream.
124  }
125  virtual ~abstract_printer() {}
126 protected:
127  // Define the requirements in an mpl::vector for containers to be printed.
128  typedef boost::mpl::vector<
129  base_and_derived<std::ios_base, _os>, // _os must derive from std::ios_base.
130  ostreamable<_os, _t>, // type _t must be ostreamable.
131  ostreamable<_os, const char*>, // C string must be ostreamable.
132  forward_iterator<_iter, const _t&>, // t range must have forward iterator.
133  same_type<_t, forward_iterator<_iter, const _t&>::value_type> // _t's type value and iterator must match.
134  > requirements;
135 
136  // Use type_erasure::any to enforce the requirements onto ostream and iterator.
137  typedef boost::type_erasure::any<requirements, _os&> ostream_type;
138 public:
139  // and iterator.
140  typedef boost::type_erasure::any<requirements, _iter> iterator_type;
141 
142 protected:
144  virtual void do_print(iterator_type first, iterator_type last, ostream_type os) const = 0;
145 }; // class abstract_printer

169 class decor_printer : public abstract_printer<>
170 {
171 public:
172  explicit decor_printer(
173  std::size_t num_columns = 0,
174  std::size_t wid = 0,
175  const std::string& pre = "\n",
176  const std::string& sep = " ", // default space as separator.
177  const std::string& suf = "\n",
178  const std::string& term = "\n")
179  :
180  cols(num_columns), width(wid), prefix(pre), separator(sep), terminator(term), suffix(suf)
181  {
187  }
188 protected:
189  // Print all items in rows.
190  virtual void do_print(iterator_type first, iterator_type last, ostream_type os) const
191  {
192  os << prefix.c_str();
193  std::size_t count = 0;
194  for(; first != last; ++first)
195  {
196  if (width > 0)
197  {
198  static_cast<std::ios_base&>(os).width(width);
199  }
200  os << *first;
201  boost::type_erasure::any<requirements, _iter> temp = first;
202  ++temp;
203  if(temp != last)
204  {
205  os << separator.c_str(); // Default space.
206  }
207  ++count;
208  if((cols != 0) && (count % cols == 0))
209  { // Last column.
210  os << suffix.c_str(); // Usually newline at end of a row.
211  }
212  }
213  os << terminator.c_str(); // Perhaps useful to have a terminating newline?
214  }
215 private:
216  std::size_t cols; // Set by constructor, number of columns (default 10).
217  std::size_t width; // Set by constructor, optional column width (default zero).
218  std::string prefix; // Set by constructor, for example, "{ ".
219  std::string separator; // Set by constructor, for example, ", ".
220  std::string suffix; // Set by constructor, for example "\n".
221  std::string terminator; // Set by constructor, for example, "}\n".
222 }; // class decor_column_printer
