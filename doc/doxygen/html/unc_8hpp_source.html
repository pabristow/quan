983  unc<is_correlated>& operator+= (const unc<is_correlated>& ud)
984  { // unc += unc
985  if (is_correlated)
986  {
987  uncertainty_ += ud.uncertainty_; // If correlated then just add.
988  degFree_ = // Choose degrees of freedom of the max uncertainty.
989  (uncertainty_ > ud.uncertainty_) ? degFree_ : ud.degFree_;
990  // Exact method of estimating not known, but is a reasonable approximation
991  // provided really is highly correlated because less well predicted
992  // variance can be predicted well with the better predicted.
993  // See email from Steve LR Ellison, Lab of Government Chem, 24 Apr 98.
994  }
995  else
996  { // Uncorrelated so Propagation of Uncertainty rules apply.
997  // Barry N Taylor & Chris E Kuyatt, NIST Technical note 1297 (1994),
998  // Guidelines for Evaluating & Expressing the Uncertainty of Measurements.
999  // http:// physics.nist.gov/pubs/guidelines.appa.html
1000  // 'Combined uncertainty' Appendix A formula A-3 = "sqrt(sum of squares)"
1001  double comb_uncertainty = sqrtSumSqrs(uncertainty_, ud.uncertainty_);
1002  degFree_ = // Welch-Satterthwaite - see NIST Appendix A3 & Appendix B-1.
1003  // Effective degrees of freedom <= - see Eq B-2, so use floor function.
1004  static_cast<unsigned short>( // Degrees of freedom must be integer by definition.
1005  floor(
1006  pow4<double>(comb_uncertainty)/
1007  (pow4<double>(uncertainty_)/degFree_ + pow4<double>(ud.uncertainty_) / ud.degFree_))
1008  );
1009  uncertainty_ = float(comb_uncertainty);
1010  }
1011  unctypes_ &= ~DEG_FREE_EXACT; // This warns of an approximation - no longer exact.
1012  unctypes_ &= ud.unctypes_; // Imposes rules for all additions of uncs.
1013  // zero + zero = zero, restrictions on pos & neg only only retained if both,
1014  // int + int is int, rational + rational is rational, int + rational is rational.
1015  // either or both plusminus same if BOTH.
1016  // Quantisation retains only if both.
1017  value_ += ud.value_;
1018  return *this;
1019  }
