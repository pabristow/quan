792  { // unc += unc
793  if (is_correlated)
794  {
795  uncertainty_ += ud.uncertainty_; // If correlated then just add.
796  degFree_ = // Choose degrees of freedom of the max uncertainty.
797  (uncertainty_ > ud.uncertainty_) ? degFree_ : ud.degFree_;
798  // Exact method of estimating not known, but is a reasonable approximation
799  // provided really is highly correlated because less well predicted
800  // variance can be predicted well with the better predicted.
801  // See email from Steve LR Ellison, Lab of Government Chem, 24 Apr 98.
802  }
803  else
804  { // Uncorrelated so Propagation of Uncertainty rules apply.
805  // Barry N Taylor & Chris E Kuyatt, NIST Technical note 1297 (1994),
806  // Guidelines for Evaluating & Expressing the Uncertainty of Measurements.
807  // http://physics.nist.gov/pubs/guidelines.appa.html
808  // 'Combined uncertainty' Appendix A formula A-3 = "sqrt(sum of squares)"
809  double comb_uncertainty = sqrtSumSqrs(uncertainty_, ud.uncertainty_);

811  // Effective degrees of freedom <= - see Eq B-2, so use floor function.
812  static_cast<unsigned short>( // Degrees of freedom must be integer by (some) definition.
813  floor(
814  pow4<double>(comb_uncertainty)/
815  (pow4<double>(uncertainty_)/degFree_ + pow4<double>(ud.uncertainty_) / ud.degFree_))
816  );
817  uncertainty_ = float(comb_uncertainty);
818  }

822  // int + int is int, rational + rational is rational, int + rational is rational.
823  // either or both plusminus same if BOTH.
824  // Quantisation retains only if both.
825  value_ += ud.value_;
