82 // Applicator is a class that stores a pointer to a function that takes
83 // a reference to an ios_base (or derived translator) argument, and
84 // an argument of the type for which the applicator is parameterized.
85 // Applicator classes have the function call operator overloaded
86 // so as to simulate a function call with the argument of the parameter type.
87 // Use, for example: oapp<int> spaces(_spaces); // allows << spaces(5) ...
88 // where function is ostream& _spaces(ostream&, int);
89 
90 // Global ostream applicator using template oapp instantiated for type int,
91 // & initialised with the addresss of function with one int parameter.
92 // Possible to use oapp<int> spaces(_spaces); which allows << spaces(5) ...
93 // but instead spaces, stars & chars done a simpler way, see S Teale p 181-3.
94 template<class T> class oapp // Applicator for ostream.
95 {
96 public:
97  oapp(ostream&(*f)(ostream&, T)) : func(f)
98  {
99  };
100  // : func(f) {} added from Watcom version to initialise & define.
101  omanip<T> operator()(T v)
102  {
103  return omanip<T>(func, v);
104  }
105 private:
106  ostream&(*func)(ostream&, T);
107 }; // class oapp

207 void outIOstates(std::ios_base::iostate rdState, ostream& os, const char* term)
208 { // Usages:
209  // Default logs cout iostate to cerr, for example "IOstate: good", or "IOstate: fail"
210  // outIOstates(); // Same as:
211  // outIOstates(cout.rdState(), cerr, ".\n");
212  // outIOstates(cin.rdState());
213  // outIOstates(cerr.rdState(), cout, ", ");
214  // outIOstates(cout.rdState(), cerr, " iostate.\n ");
215  int const savedflags = os.flags(); // Save to restore.
216  // Clear any unused and invalid bits in rdState.
217  rdState &= ios_base::goodbit | ios_base::eofbit | ios_base::failbit | ios_base::goodbit;
218  // enum _Iostate {goodbit = 0x0, eofbit = 0x1, failbit = 0x2, badbit = 0x4, _Statmask = 0x7};
219  // but use std::ios_base::goodbit; for portability.
220  // MSVC /Dinkumware defines _Statmask 0x17, _Hardfail also added recently.
221  os << "rdState ("<< showbase << hex << rdState << ") " << dec ;
222  if (rdState == ios_base::goodbit)
223  {
224  os << "good"; // 1st so no leading space.
225  }
226  else
227  { // At least one bit set, so all need a space first.
228  if ((rdState & ios_base::eofbit) != 0)
229  {
230  os << " eof";
231  }
232  if ((rdState & ios_base::failbit) != 0)
233  {
234  os << " fail";
235  }
236  if ((rdState & ios_base::badbit) != 0)
237  {
238  os << " bad";
239  }
240  }
241  os << term; // eg "\n" or ", " or " ".
242  os.flags(savedflags); // Restore.
243 } // outIOstates

277 void outFmtFlags(std::ios_base::fmtflags fmtFlags, ostream& os, const char* term)
278 {// Usage: outFmtFlags(flags, cerr);
279  // For example, logs to cerr "FormatFlags: skipws showbase right dec"
280  // Defaults in xiostream.hpp:
281  // void outFmtFlags(fmtflags fmtFlags = cout.flags(), ostream& os = cerr, const char* term = ".\n");
282  const int count = 16; // because using unsigned short int.
283  int const flags = os.flags(); // save to restore.
284  fmtFlags &= 0x7FFF; // _Fmtmask // clear un-used bits.
285  os << "iosFormatFlags (" << showbase << hex << fmtFlags << std::dec << ")" ; // hex value.
286  if (fmtFlags != 0)
287  {
288  for(int i = 0, j = 1; i < count; ++i)
289  {
290  if ((fmtFlags & j) != 0)

292  os << ' ' << fmtFlagWords[i];

294  j <<= 1;
295  }

340 void outFpClass(double value, ostream& os = cerr)
341 // Usage: outFpClass(x, cout);
342 { // Custom outputs for NaN, inf ... (rather than default 1#IND ...)
343  int fpclass = _fpclass(value);
344  switch (fpclass)
345  { // if !_finite true then FPclass values defined in <cfloat>
346  case _FPCLASS_SNAN: // Signaling NaN
347  cerr << "Signaling NaN! " << endl;
348  os << "?" ;
349  break;
350  case _FPCLASS_QNAN: // Quiet NaN
351  cerr << "Quiet NaN." << endl;
352  os << "? " ;
353  break;
354  case _FPCLASS_NINF: // Negative infinity ( INF)
355  // std::numeric_limits<double>::signaling_Nan() seems to
356  // code as -1#INF rather than 1#IND as quiet_Nan()
357  // cerr << "Negative infinity!" << endl;
358  // os << "-infinity " ;
359  cerr << "Signaling NaN! " << endl;
360  os << "? " ;
361  break;
362  case _FPCLASS_NN: // Negative normalized non-zero
363  cerr << "Negative normalised non-zero!" << endl;
364  break;
365  case _FPCLASS_ND: // Negative denormalized
366  cerr << "Negative denormalized!" << endl;
367  break;
368  case _FPCLASS_NZ: // Negative zero ( 0) ? is finite?
369  cerr << "Negative zero!" << endl;
370  os << "-0"; // Should never get here?
371  break;
372  case _FPCLASS_PZ: // Positive zero ( + 0) - but is finite!
373  cerr << "Positive zero!" << endl;
374  os << "0 "; // Should never get here!
375  break;
376  case _FPCLASS_PN: // Positive normalized non-zero - but is finite!
377  cerr << "Positive normalized non-zero" << endl;
378  os << value ; // Should never get here! should be output normally.
379  break;
380  case _FPCLASS_PINF: // Positive infinity (+INF)
381  cerr << "Positive infinity (+INF)" << endl;
382  os << "inf" ;
383  break;
384  default: // Unexpected value for fpclass!
385  cerr << "\nFloating point value " << value <<
386  ", unexpected _fpclass = " << _fpclass(value) << endl;
387  } //switch
388 } // OutFpclass
