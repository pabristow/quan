00671 unsigned int findUnit(const string multipleUnit,
00672                       // "mm", "Mm", "m", "um" "centim", "centimeter"...
00673                       const unit* u, // Unit type: mass, length, current etc
00674                       unsigned int& givenIndex,  // index to uNames
00675                       // unit name (or abbreviation or alias) given,
00676                       // for example " for inch.
00677                                                                                         // else notOK if unexpected.
00678                       unsigned int& SIindex) // index to SIsymbols & SIprefixes
00679                                                                                         // else notOK if unexpected prefix.
00680                                                                                         // Or SImaxPowerTen if not scaled?
00681                       
00682                                                                                         // return uNamesIndex to index u->unitNames[uNamesIndex];
00683                                                                                         // meter, inch, foot ...
00684                                                                                         // & u->unitToSIfactors[uNamesIndex] else -1 aka notOK
00685 {
00686   bool isSIsymbol = false; // true if is k, m, M ...
00687   bool isSIprefix = false;  // true if is kilo, mill, Mega ...
00688   bool isSImultiple = false;  // if SIsymbol or SIprefix
00689   bool isAmbig = false;  // if SImultiple AND isUnit, eg m = milli and meter.
00690   bool isUnitAbbrev = false;  // true if unit abbreviation like m, Hz found.
00691   bool isUnit = false;  // true if a full unit like meter ... found.
00692   bool isSIunit = false;  // true if meter or second but false if foot or minute.
00693   //int uNamesIndex = 0;  // if == 0, then SI unit, else another unit like inch.
00694   unsigned int uNamesIndex = notOK;
00695         // if == 0 SI base name, if == 1 SI abbreviation, else an alias or plural.
00696   // value to return.
00697   unsigned int SIprefixLength = 0;  // Example: length of "kilo" = 4.
00698   unsigned int SIsymbolLength; // length of SI symbol = 1, except "da".
00699   unsigned int symbolOrPrefixLength = 0;
00700   int unitNameLength = 0;  // length of unit name.
00701         unsigned int unitPos = 0; // position of unit string in multipleUnit.
00702   string SIsymbol = "";  // Example 'k'.
00703   string SIprefix = "";  // Example "kilo".
00704   string unitName = "";   // Base unit name, for example meter, inch, amp oz.
00705   string suffixUnitName = "";  // What is left after SI prefix or symbol removed.
00706   string SIunitName = ""; // SI unit for type of unit, meter for length.
00707   string givenUnitName = "";  // = uNames[givenIndex];
00708         string prefixOrSymbol = "";
00709   const char* unitAbbreviation = "";
00710   SIindex = notOK;  // index to SIsymbols, SIprefixes & powersOfTen.
00711   double confactor = 1.0; // Multiply by this to convert to SI unit.
00712   
00713   if (multipleUnit.length() == 0) 
00714   {
00715     cerr << "No unit (or multiple) given!" << endl;
00716     return notOK;
00717   }
00718   
00719   // To distinguish mm from m:
00720   // if 1st char != a SIsymbol then must just be unit.
00721   
00722   // If a single char, then must be just unit and not SIsymbol.
00723   // For example, just m must be m for meter, and not m for milli,
00724   // A must be Amp not atto ...
00725   // Cases like mm = millimeter and aA attoAmpere interesting!
00726   // This will be caught by check for unit abbreviations.
00727   // Problem if mm cos both multiple AND unit, if use search order
00728   // unit, prefix or symbol, check for unit will be found OK,
00729   // even if in the 'wrong' order (first m = meter and second = milli).
00730   // Need to check that unit found OK, or could be confused with
00731   // just a SI prefix or symbol and no unit given.
00732   
00733   // First check if just unit like meter, etc without symbol or prefix.
00734   // or abbreviation like m (for meter) or A (for ampere),
00735         // or other non-SI unit like feet, inch, mile ...
00736         // Note that singular meter is before plural meters, so givenName = meter
00737         // even if enter meters, or inch if enter inches.
00738         // Further complication is that Celsius ends with s, a unit abbreviation,
00739         // so finding unit abbreviation at end is not enough, multiple must match.
00740         
00741   // cout << ' ' << int(u->unitNamesCount) << endl;  // To show total
00742   // names, abbreviations & aliases for this type of unit, like length.
00743         
00744   for (uNamesIndex = 0;
00745                 uNamesIndex != unsigned int(u->unitNamesCount);
00746                 ++uNamesIndex)
00747         {
00748                 int ui = uNamesIndex;
00749                 const char** uNames = u->unitNames[uNamesIndex];
00750                 // Array of length units like meter, feet, mile, inch, point ...
00751                 //cerr << "uNames[uNamesIndex] " << uNames[0] << endl;
00752                 
00753                 for (int namesIndex = 0;
00754                         namesIndex != u->unitLengths[uNamesIndex];
00755                         // Count of aliases for this unit type.
00756                         namesIndex++)
00757                 {
00758                         string uName = uNames[namesIndex];
00759                         // An alias or abbreviation like m, meter, meters, metre ...
00760                         //cerr << "uName " << uName.c_str() << endl;
00761                         unitNameLength = uName.length();
00762                         if (unitNameLength <= 0)
00763                                 continue;  // Null string names like unknown never matches.
00764                         unitPos = multipleUnit.rfind(uName);
00765                         // Reverse find unit name in multipleUnit.
00766                         if (unitPos == string::npos)
00767                                 continue;  // unit name uName not found at all in multipleUnit.
00768                         isSIunit = (0 == uNamesIndex) ? true : false;  // If inch not meter.
00769                         givenIndex = namesIndex;
00770                         // givenUnitName = multipleUnit.substr(unitPos, unitNameLength);
00771                         givenUnitName = multipleUnit.substr(unitPos);  // to end.
00772                         //cerr << " Given " << givenUnitName.c_str() << ", uName " << uName.c_str() << endl;
00773                         if (givenUnitName.compare(uName) != 0)
00774                                 continue; // Doesn't match unit name fully.
00775                         isUnit = true;  // but findMultiple might fail.
00776                         unitName = uNames[0];
00777                         // Base unit name like inch, not abbreviation or alias, is 1st in array.
00778                         unitAbbreviation = uNames[1];  // SI Abbreviation is 2nd in array.
00779                         SIunitName = u->SIunitNames[0];  // For example: meter, gram, second ... 
00780                         confactor = u->unitToSIfactors[uNamesIndex]; // Conversion factor to SI unit.
00781                         if (0 == unitPos) // Found at start.
00782                         {  // Found ONLY a unit name, unit abbreviation or unit alias.
00783                                 isSImultiple = false;   isSIsymbol = false;     isSIprefix = false;
00784                                 return uNamesIndex;     // Without checking for any prefix or symbol.
00785                         }
00786                         else
00787                         { // Found at end, so may be multiple prefix or symbol, or mismatch.
00788                                         // findMultiple: // To try to find multiple symbol or prefix.
00789                         prefixOrSymbol = multipleUnit.substr(0, unitPos);
00790                                 cerr << "prefixOrSymbol " << prefixOrSymbol.c_str() << endl;
00791                                 isSImultiple = false; isSIsymbol = false; isSIprefix = false;
00792                                 SIindex = notOK;  // Assume doesn't start with SIsymbol or prefix.
00793                                 // Check if starts with a multiple symbol (milli) or prefix (m).
00794                                 for (int SIIndex = 0; SIIndex != SIprefixesLength; ++SIIndex)
00795                                 {  // Try all the SI multiples in turn.
00796                                         SIsymbol = SIsymbols[SIIndex];  // Example: "y", ... "k" ...
00797                                         SIprefix = SIprefixes[SIIndex];  // Example: "yocto" ... "kilo".
00798                                         SIprefixLength = SIprefix.length();
00799                                         SIsymbolLength = SIsymbol.length();
00800                                         if (0 == SIsymbolLength)
00801                                                 continue;  // SIsymbol is "", so prefix known to be "" too.
00802                                 // Check for prefix "kilo" before "k" or will leave "ilo" behind!
00803                                         if (0 == prefixOrSymbol.find(SIprefix) )
00804                                         {       // So matched a prefix like kilo (not just a symbol like k).
00805                                                 isSIprefix = true;  // & SIprefix holds which prefix.
00806                                                 symbolOrPrefixLength = SIprefixLength;
00807                                                 SIindex = SIIndex;
00808                                                 break;
00809                                         }  // find == 0
00810                                         if (prefixOrSymbol.length() <= 2 && 0 == prefixOrSymbol.find(SIsymbol) )
00811                                         {       // Matched a symbol like k or da, (but didn't find a prefix like kilo).
00812                                                 isSIsymbol = true;  // & SIprefix holds which prefix too.
00813                                                 symbolOrPrefixLength = SIsymbolLength;
00814                                                 SIindex = SIIndex;
00815                                                 break;
00816                                         }   // 0 == find
00817                                 }  // for SIIndex...
00818                                 isSImultiple = isSIsymbol || isSIprefix;
00819                                 if (!isSImultiple)
00820                                 {
00821                                         isUnit = false;
00822                                         cerr << "Unit " << givenUnitName.c_str() << ", but symbol or prefix " 
00823                                                 << prefixOrSymbol.c_str() << " not recognised" << endl;
00824                                         // May need to search more,
00825                                         // for example: Celsius ends with s abbreviation for unit of time.
00826                                         continue;
00827                                 }
00828                                 else
00829                                 {  // multiple & Unit both found OK.
00830                                         // cerr << "uNamesIndex " << uNamesIndex << ", SIindex " << SIindex << endl;
00831                                         return uNamesIndex;  
00832                                 } // isSImultiple
00833                         }  // found at start or end
00834                 }  // for namesIndex : meter, m, meters...
00835         } // for uNamesIndex. // meter, mile, yard, feet, inch ...
00836         cerr << suffixUnitName.c_str() << " is NOT a recognised unit of "
00837                 << u->unitOf << '!' << endl;  // for example length, mass ...
00838         unitName = suffixUnitName;
00839         isSImultiple = false;
00840         return notOK;
00841 }  // findUnit
