00082 // Applicator is a class that stores a pointer to a function that takes
00083 // a reference to an ios_base (or derived translator) argument, and
00084 // an argument of the type for which the applicator is parameterized.
00085 // Applicator classes have the function call operator overloaded
00086 // so as to simulate a function call with the argument of the parameter type.
00087 // Use, for example: oapp<int> spaces(_spaces);  // allows << spaces(5) ...
00088 // where function is ostream& _spaces(ostream&, int);
00089 
00090 // Global ostream applicator using template oapp instantiated for type int,
00091 // & initialised with the addresss of function with one int parameter.
00092 // Possible to use oapp<int> spaces(_spaces);  which allows << spaces(5) ...
00093 // but instead spaces, stars & chars done a simpler way, see S Teale p 181-3.
00094 template<class T> class oapp  // Applicator for ostream.
00095 {
00096 public:
00097   oapp(ostream&(*f)(ostream&, T)) : func(f)
00098   {
00099   };
00100   //  : func(f) {} added from Watcom version to initialise & define.
00101   omanip<T> operator()(T v)
00102   {
00103     return omanip<T>(func, v);
00104   }
00105 private:
00106   ostream&(*func)(ostream&, T);
00107 }; // class oapp

00207 void outIOstates(std::ios_base::iostate rdState, ostream& os, const char* term)
00208 { // Usages:
00209   // Default logs cout iostate to cerr, for example "IOstate: good", or "IOstate: fail"
00210   // outIOstates(); // Same as:
00211   // outIOstates(cout.rdState(), cerr, ".\n");
00212   // outIOstates(cin.rdState());
00213   // outIOstates(cerr.rdState(), cout, ", ");
00214   // outIOstates(cout.rdState(), cerr, " iostate.\n ");
00215   int const savedflags = os.flags();  // Save to restore.
00216   // Clear any unused and invalid bits in rdState.
00217   rdState &= ios_base::goodbit | ios_base::eofbit | ios_base::failbit | ios_base::goodbit;
00218   // enum _Iostate {goodbit = 0x0, eofbit = 0x1,            failbit = 0x2, badbit = 0x4, _Statmask = 0x7};
00219   // but use std::ios_base::goodbit; for portability.
00220   // MSVC /Dinkumware defines _Statmask 0x17, _Hardfail also added recently.
00221   os << "rdState ("<< showbase << hex << rdState << ") " << dec ;
00222   if (rdState == ios_base::goodbit)
00223   {
00224     os << "good"; // 1st so no leading space.
00225   }
00226   else
00227   { // At least one bit set, so all need a space first.
00228     if ((rdState & ios_base::eofbit) != 0)
00229     {
00230       os << " eof";
00231     }
00232     if ((rdState & ios_base::failbit) != 0)
00233     {
00234       os << " fail";
00235     }
00236     if ((rdState & ios_base::badbit) != 0)
00237     {
00238       os << " bad";
00239     }
00240   }
00241   os << term; // eg "\n" or ", " or " ".
00242   os.flags(savedflags); // Restore.
00243 }  // outIOstates
