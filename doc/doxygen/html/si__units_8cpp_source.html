599 void showAllUnits(const unit* u, ostream& os) // u = &mass, &length, time ...
600 {
601  int allNamesCount = 0; // Count all names for this unit.
602  for (int ic = 0; ic < u->unitNamesCount; ic++)
603  {
604  allNamesCount += u->unitLengths[ic];
605  }
606  // & check that this tallies with the stored value.
607  // Run this function to peform this check after any changes!
608  if (allNamesCount != u->unitsNamesCount)
609  os << u->unitOf << ": sum of units counts is " << allNamesCount
610  << ", but unitsNamesCount is " << int(u->unitsNamesCount)
611  << "!" << endl;
612  showSIunits(u, os); // which calls showSIunit.
613  // showNonSIunits(u, os);
614  showAllNonSIunits(u, os);
615  os << endl;

688 unsigned int findUnit(const string multipleUnit,
689  // "mm", "Mm", "m", "um" "centim", "centimeter"...
690  const unit* u, // Unit type: mass, length, current etc
691  unsigned int& givenIndex, // index to uNames
692  // unit name (or abbreviation or alias) given,
693  // for example " for inch.
694  // else notOK if unexpected.
695  unsigned int& SIindex) // index to SIsymbols & SIprefixes
696  // else notOK if unexpected prefix.
697  // Or SImaxPowerTen if not scaled?
698 
699  // return uNamesIndex to index u->unitNames[uNamesIndex];
700  // meter, inch, foot ...
701  // & u->unitToSIfactors[uNamesIndex] else -1 aka notOK
702 {
703  bool isSIsymbol = false; // true if is k, m, M ...
704  bool isSIprefix = false; // true if is kilo, milli, Mega ...
705  bool isSImultiple = false; // if SIsymbol or SIprefix
706  // bool isAmbig = false; // if SImultiple AND isUnit, eg m = milli and meter.
707  // bool isUnitAbbrev = false; // true if unit abbreviation like m, Hz found.
708  bool isUnit = false; // true if a full unit like meter ... found.
709  bool isSIunit = false; // true if meter or second but false if foot or minute.
710  //int uNamesIndex = 0; // if == 0, then SI unit, else another unit like inch.
711  unsigned int uNamesIndex = notOK;
712  // if == 0 SI base name, if == 1 SI abbreviation, else an alias or plural.
713  // value to return.
714  size_t SIprefixLength = 0; // Example: length of "kilo" = 4.
715  size_t SIsymbolLength; // length of SI symbol = 1, except "da".
716  size_t symbolOrPrefixLength = 0;
717  size_t unitNameLength = 0; // length of unit name.
718  size_t unitPos = 0; // position of unit string in multipleUnit.
719  string SIsymbol = ""; // Example 'k'.
720  string SIprefix = ""; // Example "kilo".
721  string unitName = ""; // Base unit name, for example meter, inch, amp oz.
722  string suffixUnitName = ""; // What is left after SI prefix or symbol removed.
723  string SIunitName = ""; // SI unit for type of unit, meter for length.
724  string givenUnitName = ""; // = uNames[givenIndex];
725  string prefixOrSymbol = "";
726  const char* unitAbbreviation = "";
727  SIindex = notOK; // index to SIsymbols, SIprefixes & powersOfTen.
728  double confactor = 1.0; // Multiply by this to convert to SI unit.
729 
730  if (multipleUnit.length() == 0)
731  {
732  cerr << "No unit (or multiple) given!" << endl;
733  return notOK;
734  }
735 
736  // To distinguish mm from m:
737  // if 1st char != a SIsymbol then must just be unit.
738 
739  // If a single char, then must be just unit and not SIsymbol.
740  // For example, just m must be m for meter, and not m for milli,
741  // A must be Amp not atto ...
742  // Cases like mm = millimeter and aA attoAmpere interesting!
743  // This will be caught by check for unit abbreviations.
744  // Problem if mm cos both multiple AND unit, if use search order
745  // unit, prefix or symbol, check for unit will be found OK,
746  // even if in the 'wrong' order (first m = meter and second = milli).
747  // Need to check that unit found OK, or could be confused with
748  // just a SI prefix or symbol and no unit given.
749 
750  // First check if just unit like meter, etc without symbol or prefix.
751  // or abbreviation like m (for meter) or A (for ampere),
752  // or other non-SI unit like feet, inch, mile ...
753  // Note that singular meter is before plural meters, so givenName = meter
754  // even if enter meters, or inch if enter inches.
755  // Further complication is that Celsius ends with s, a unit abbreviation,
756  // so finding unit abbreviation at end is not enough, multiple must match.
757 
758  // cout << ' ' << int(u->unitNamesCount) << endl; // To show total
759  // names, abbreviations & aliases for this type of unit, like length.
760 
761  for (uNamesIndex = 0;
762  uNamesIndex != unsigned int(u->unitNamesCount);
763  ++uNamesIndex)
764  {
765  //int ui = uNamesIndex;
766  const char** uNames = u->unitNames[uNamesIndex];
767  // Array of length units like meter, feet, mile, inch, point ...
768  //cerr << "uNames[uNamesIndex] " << uNames[0] << endl;
769 
770  for (int namesIndex = 0;
771  namesIndex != u->unitLengths[uNamesIndex];
772  // Count of aliases for this unit type.
773  namesIndex++)
774  {
775  string uName = uNames[namesIndex];
776  // An alias or abbreviation like m, meter, meters, metre ...
777  //cerr << "uName " << uName.c_str() << endl;
778  unitNameLength = uName.length();
779  if (unitNameLength <= 0)
780  continue; // Null string names like unknown never matches.
781  unitPos = multipleUnit.rfind(uName);
782  // Reverse find unit name in multipleUnit.
783  if (unitPos == string::npos)
784  continue; // unit name uName not found at all in multipleUnit.
785  isSIunit = (0 == uNamesIndex) ? true : false; // If inch not meter.
786  givenIndex = namesIndex;
787  // givenUnitName = multipleUnit.substr(unitPos, unitNameLength);
788  givenUnitName = multipleUnit.substr(unitPos); // to end.
789  //cerr << " Given " << givenUnitName.c_str() << ", uName " << uName.c_str() << endl;
790  if (givenUnitName.compare(uName) != 0)
791  continue; // Doesn't match unit name fully.
792  isUnit = true; // but findMultiple might fail.
793  unitName = uNames[0];
794  // Base unit name like inch, not abbreviation or alias, is 1st in array.
795  unitAbbreviation = uNames[1]; // SI Abbreviation is 2nd in array.
796  SIunitName = u->SIunitNames[0]; // For example: meter, gram, second ...
797  confactor = u->unitToSIfactors[uNamesIndex]; // Conversion factor to SI unit.
798  if (0 == unitPos) // Found at start.
799  { // Found ONLY a unit name, unit abbreviation or unit alias.
800  isSImultiple = false; isSIsymbol = false; isSIprefix = false;
801  return uNamesIndex; // Without checking for any prefix or symbol.
802  }
803  else
804  { // Found at end, so may be multiple prefix or symbol, or mismatch.
805  // findMultiple: // To try to find multiple symbol or prefix.
806  prefixOrSymbol = multipleUnit.substr(0, unitPos);
807  cerr << "prefixOrSymbol " << prefixOrSymbol.c_str() << endl;
808  isSImultiple = false; isSIsymbol = false; isSIprefix = false;
809  SIindex = notOK; // Assume doesn't start with SIsymbol or prefix.
810  // Check if starts with a multiple symbol (milli) or prefix (m).
811  for (int SIIndex = 0; SIIndex != SIprefixesLength; ++SIIndex)
812  { // Try all the SI multiples in turn.
813  SIsymbol = SIsymbols[SIIndex]; // Example: "y", ... "k" ...
814  SIprefix = SIprefixes[SIIndex]; // Example: "yocto" ... "kilo".
815  SIprefixLength = SIprefix.length();
816  SIsymbolLength = SIsymbol.length();
817  if (0 == SIsymbolLength)
818  continue; // SIsymbol is "", so prefix known to be "" too.
819  // Check for prefix "kilo" before "k" or will leave "ilo" behind!
820  if (0 == prefixOrSymbol.find(SIprefix) )
821  { // So matched a prefix like kilo (not just a symbol like k).
822  isSIprefix = true; // & SIprefix holds which prefix.
823  symbolOrPrefixLength = SIprefixLength;
824  SIindex = SIIndex;
825  break;
826  } // find == 0
827  if (prefixOrSymbol.length() <= 2 && 0 == prefixOrSymbol.find(SIsymbol) )
828  { // Matched a symbol like k or da, (but didn't find a prefix like kilo).
829  isSIsymbol = true; // & SIprefix holds which prefix too.
830  symbolOrPrefixLength = SIsymbolLength;
831  SIindex = SIIndex;
832  break;
833  } // 0 == find
834  } // for SIIndex...
835  isSImultiple = isSIsymbol || isSIprefix;
836  if (!isSImultiple)
837  {
838  isUnit = false;
839  cerr << "Unit " << givenUnitName.c_str() << ", but symbol or prefix "
840  << prefixOrSymbol.c_str() << " not recognised" << endl;
841  // May need to search more,
842  // for example: Celsius ends with s abbreviation for unit of time.
843  continue;
844  }
845  else
846  { // multiple & Unit both found OK.
847  // cerr << "uNamesIndex " << uNamesIndex << ", SIindex " << SIindex << endl;
848  return uNamesIndex;
849  } // isSImultiple
850  } // found at start or end
851  } // for namesIndex : meter, m, meters...
852  } // for uNamesIndex. // meter, mile, yard, feet, inch ...
853  cerr << suffixUnitName.c_str() << " is NOT a recognised unit of "
854  << u->unitOf << '!' << endl; // for example length, mass ...
855  unitName = suffixUnitName;
856  isSImultiple = false;
857  return notOK;
858 } // findUnit
